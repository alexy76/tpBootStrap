git init

git remote add origin https://.....git

//Vérifier le remote
git remote -v

// Remplacement du remote
git remote set-url origin  http://....git

//Fichier suivi
git status

//Indexer les fichiers
git add fichier
git add -A   /    .    / --a

// configurer les globales
git config --global user.mail "mail"
git config --global user.name "name"

//Pour commiter
git commit -am "commentaire pertinent"

//Verifier les commits
git log

//Ouvrir un fichier pour edition sur vscode
code fichier.html

//Envoyer sur le repo distant
git push origin master(main)

//Modifier le dernier commentaire du commit
git commit --amend -m "nouveau commentaire"

//Récupérer l'intégralité d'un repo
git clone http://github.com/pseudo/repo

//Renommer le remote du repo
git remote rename <old> <new>

//Créer une branche en local puis sur le serveur github
*git checkout -b sousTitre*
git branch sousTitre

git push origin sousTitre

//Se déplacer sur une autre branche
git checkout nom_de_la_branche

//Mettre à jour le repo local
git pull origin master

//Fusioner des branches entre elles
git merge sousTitre (quand on se trouve dans la branche master)

//Supprimer une branche local
git branch -d sousTitre

//Supprimer une branche sur le github
git push origin --delete sousTitre

//Pour annuler une fusion local
git reset --merge
